// SPDX-License-Identifier: MIT pragma solidity ^0.8.20; import "@openzeppelin/contracts/token/ERC20/ERC20.sol"; import "@openzeppelin/contracts/access/Ownable.sol"; contract AlFareedOfficeToken is ERC20, Ownable { // نسبة المكافأة الافتراضية (مثلاً 1% من قيمة المعاملة) uint256 public rewardRate = 1; // رسم تعبيري لمناطق الخدمة (6 أكتوبر والشيخ زايد) لتوثيق النطاق الجغرافي string public constant SERVICE_AREA = "6th of October & Sheikh Zayed"; event ServicesRewardSent(address indexed customer, uint256 amount); constructor() ERC20("Al-Fareed Office Token", "FARID") Ownable(msg.sender) { // صك كمية أولية للمالك (أو لحوض السيولة وفقاً لـ PiRC1) _mint(msg.sender, 1000000 * 10 ** decimals()); } // وظيفة تسجيل طلب خدمة ومنح مكافأة (تطبيقاً لمبدأ المنفعة في PiRC1) function completeService(address customer, uint256 orderValue) public onlyOwner { require(customer != address(0), "Invalid customer address"); uint256 rewardAmount = (orderValue * rewardRate) / 100; // التحقق من توفر رصيد في العقد للمكافآت require(balanceOf(owner()) >= rewardAmount, "Insufficient rewards in treasury"); _transfer(owner(), customer, rewardAmount); emit ServicesRewardSent(customer, rewardAmount); } // تحديث نسبة المكافآت حسب العروض الموسمية function updateRewardRate(uint256 newRate) public onlyOwner { rewardRate = newRate; } // وظيفة تسمح للمستخدمين بحرق (Burn) التوكنز مقابل خصم على الخدمات الورقية أو الخرسانة function redeemForDiscount(uint256 amount) public { _burn(msg.sender, amount); // هنا يتم تسجيل الخصم في قاعدة بيانات المكتب الخارجية } }
