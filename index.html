// استدعاء المكتبة const Pi = window.Pi; // تهيئة التطبيق - تأكد من استخدام sandbox: true للمرحلة التجريبية Pi.init({ version: "2.0", sandbox: true }); async function loginWithPi() { const loginBtn = document.getElementById('login-btn'); try { // تغيير نص الزر أثناء التحميل loginBtn.innerText = "جاري الاتصال..."; const scopes = ['username', 'payments']; // دالة التعامل مع الدفعات المعلقة (متطلب أساسي لـ Pi) function onIncompletePaymentFound(payment) { console.log("Incomplete payment found", payment); }; // طلب التوثيق const auth = await Pi.authenticate(scopes, onIncompletePaymentFound); // عند النجاح console.log("Success:", auth); alert(`أهلاً بك يا ${auth.user.username} في تطبيق مكتب الفريد`); // يمكنك هنا توجيه المستخدم لصفحة خدمات الخرسانة loginBtn.innerText = "تم تسجيل الدخول"; loginBtn.style.backgroundColor = "#4CAF50"; } catch (err) { // في حال الخطأ أو فتح المتصفح من خارج Pi Browser console.error(err); alert("يرجى فتح التطبيق من داخل Pi Browser لتفعيل خدمات مكتب الفريد."); loginBtn.innerText = "فشل الدخول - حاول مجدداً"; } }
