// تهيئة الـ SDK Pi.init({ version: "2.0", sandbox: true }); async function login() { try { // نطلب اليوزر نيم فقط في البداية لضمان سرعة الاستجابة const auth = await Pi.authenticate(['username'], function(payment) { console.log("Payment in progress", payment); }); // أول ما ينجح الربط، نظهر واجهة الطلب ونخفي زر الدخول document.getElementById('login-area').style.display = 'none'; document.getElementById('user-section').style.display = 'block'; document.getElementById('username').innerText = auth.user.username; console.log("تم الربط بنجاح يا برنس باسم: " + auth.user.username); } catch (err) { // لو حصل خطأ، نظهر تنبيه للمستخدم console.error(err); alert("عفواً، تأكد من فتح التطبيق من داخل Pi Browser وتفعيل وضع Sandbox"); } }
