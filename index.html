async function login() { const status = document.getElementById("status"); status.innerHTML = "جاري التحقق.. من فضلك انتظر"; try { // يجب تحديد الصلاحيات (Scopes) التي يطلبها التطبيق const scopes = ['username']; // دالة ضرورية للتعامل مع أي مدفوعات معلقة (حتى لو لم تستخدمها الآن) const onIncompletePaymentFound = (payment) => { console.log("Incomplete payment found", payment); }; const auth = await window.Pi.authenticate(scopes, onIncompletePaymentFound); status.innerHTML = "✅ أهلاً بك: " + auth.user.username; status.style.color = "green"; } catch (e) { status.innerHTML = "❌ خطأ: يرجى فتح التطبيق من متصفح Pi"; status.style.color = "red"; console.error(e); } }
